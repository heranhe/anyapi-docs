info:

                    原图左边距离新图左边为30 x 0.3 = 9，区域内均为扩图范围
                right_expansion_ratio:
                  type: number
                  description: |-
                    向右扩充范围；基于原图宽度的倍数而计算
                    取值范围：[0,2]，新图片整体面积不得超过原图片3倍
                    如原图宽30，当前参数值为0.4，则：
                    原图右边距离新图右边为30 x 0.4 = 12，区域内均为扩图范围
                prompt:
                  type: string
                  description: |-
                    正向文本提示词
                    不能超过2500个字符
                'n':
                  type: integer
                  description: |-
                    生成图片数量
                    取值范围：[1,9]
                callback_url:
                  type: string
                external_task_id:
                  type: string
